-- Business Goal: Find Top 10 High-Value Customers for the Year 2024
-- Optimization: Using CTEs to reduce the dataset before the final join

WITH CustomerTotal AS (
    SELECT 
        u.user_id,
        u.username,
        SUM(oi.price) as total_spent,
        COUNT(o.order_id) as total_orders
    FROM Users u
    INNER JOIN Orders o ON u.user_id = o.user_id
    INNER JOIN Order_Items oi ON o.order_id = oi.order_id
    WHERE o.order_date >= '2024-01-01' -- Using index on order_date
    GROUP BY u.user_id, u.username
)
-- Final Selection with sorting
SELECT 
    user_id, 
    username, 
    total_spent, 
    total_orders
FROM CustomerTotal
WHERE total_spent > 1000
ORDER BY total_spent DESC
LIMIT 10;
